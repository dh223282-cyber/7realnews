service cloud.firestore {
  match /databases/{database}/documents {
    match /posts/{post} {
      allow read: if true;
      allow write: if request.auth != null; 
    }
    match /news/{newsItem} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    match /advertisements/{ad} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    match /ads/{ad} {
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
}
